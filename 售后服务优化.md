# 售后服务优化

## 一、理论框架

### 1.1 售后服务的战略价值

售后服务是客户生命周期中建立忠诚度的关键环节：

**售后服务数据洞察：**
- 93%的消费者会因优质服务成为回头客
- 解决投诉的客户70%会再次购买
- 获客成本是留存成本的5-7倍
- 提升5%留存率可增加25-95%利润

### 1.2 售后服务类型

**退换货服务：**
- 无理由退货
- 质量问题换货
- 尺码/颜色调换
- 退款处理

**维修服务：**
- 保修期内维修
- 付费维修服务
- 零配件更换
- 延保服务

**咨询服务：**
- 产品使用指导
- 订单状态查询
- 投诉处理
- 建议收集

**增值服务：**
- 以旧换新
- 产品升级
- 会员专享服务
- VIP客服通道

### 1.4 售后服务KPI体系

| KPI | 定义 | 目标值 |
|-----|------|--------|
| 首次解决率(FCR) | 一次接触解决问题比例 | >80% |
| 平均响应时间 | 客户等待时间 | <5分钟 |
| 平均解决时间 | 问题从发起到解决 | <24小时 |
| 客户满意度(CSAT) | 服务评分 | >4.5/5 |
| 退货率 | 退货订单比例 | <5% |
| 退货处理时效 | 退款到账时间 | <3天 |

```javascript
class ReturnHandler {
  constructor() {
    this.statuses = {
      pending: '待审核',
      approved: '已批准',
      rejected: '已拒绝',
      returning: '退货中',
      received: '已收货',
      refunded: '已退款'
    };
  }
  
  async createReturnRequest(data) {
    // 1. 验证订单
    const order = await this.validateOrder(data.orderId);
    if (!order) throw new Error('订单不存在');
    
    // 2. 检查退货期限
    if (!this.isWithinReturnWindow(order)) {
      throw new Error('已超过退货期限');
    }
    
    // 3. 创建退货工单
    const returnRequest = await this.createRequest({
      orderId: data.orderId,
      items: data.items,
      type: data.type,
      reason: data.reason,
      description: data.description,
      images: data.images,
      status: 'pending'
    });
    
    // 4. 发送确认通知
    await this.sendNotification(returnRequest, 'created');
    
    return returnRequest;
  }
  
  async approveRequest(requestId) {
    const request = await this.getRequest(requestId);
    
    // 更新状态
    request.status = 'approved';
    await this.saveRequest(request);
    
    // 生成退货标签
    const returnLabel = await this.generateReturnLabel(request);
    
    // 通知客户
    await this.sendNotification(request, 'approved', { returnLabel });
  }
  
  async processReceivedItem(requestId, trackingNumber) {
    const request = await this.getRequest(requestId);
    
    // 更新状态
    request.status = 'received';
    request.receivedAt = new Date();
    await this.saveRequest(request);
    
    // 触发质检流程
    await this.triggerQualityCheck(request);
  }
  
  async processRefund(requestId) {
    const request = await this.getRequest(requestId);
    
    // 计算退款金额
    const refundAmount = this.calculateRefund(request);
    
    // 执行退款
    const refund = await this.executeRefund({
      orderId: request.orderId,
      amount: refundAmount,
      method: request.refundMethod
    });
    
    // 更新状态
    request.status = 'refunded';
    request.refundedAt = new Date();
    request.refundId = refund.id;
    await this.saveRequest(request);
    
    // 发送通知
    await this.sendNotification(request, 'refunded');
    
    // 恢复库存
    await this.restoreInventory(request.items);
  }
  
  calculateRefund(request) {
    let amount = request.items.reduce((sum, item) => sum + item.price, 0);
    
    // 检查是否扣除运费
    if (request.reason === 'changed_mind') {
      amount -= request.originalShipping || 0;
    }
    
    return amount;
  }
}
```

```javascript
class TicketSystem {
  constructor() {
    this.priorities = {
      low: { name: '低', sla: 48 },
      medium: { name: '中', sla: 24 },
      high: { name: '高', sla: 8 },
      urgent: { name: '紧急', sla: 2 }
    };
  }
  
  async createTicket(data) {
    // 自动分类
    const category = await this.classifyTicket(data);
    
    // 确定优先级
    const priority = this.determinePriority(data, category);
    
    // 分配处理人
    const assignee = await this.assignAgent(category);
    
    const ticket = {
      id: this.generateId(),
      customerId: data.customerId,
      subject: data.subject,
      description: data.description,
      category: category,
      priority: priority,
      status: 'open',
      assignee: assignee,
      createdAt: new Date(),
      slaDeadline: this.calculateSLA(priority),
      messages: [{
        from: 'customer',
        content: data.description,
        time: new Date()
      }]
    };
    
    await this.saveTicket(ticket);
    
    // 发送通知
    await this.notifyAgent(assignee, ticket);
    
    return ticket;
  }
  
  determinePriority(data, category) {
    // VIP客户自动高优先级
    if (data.customerTier === 'vip') return 'high';
    
    // 投诉类自动高优先级
    if (category === 'complaint') return 'high';
    
    // 退款问题中等优先级
    if (category === 'refund') return 'medium';
    
    return 'low';
  }
  
  async assignAgent(category) {
    // 获取在线且擅长该类别的客服
    const agents = await this.getAvailableAgents(category);
    
    // 按当前工单数量排序，分配给最少的
    agents.sort((a, b) => a.activeTickets - b.activeTickets);
    
    return agents[0]?.id || null;
  }
  
  calculateSLA(priority) {
    const hours = this.priorities[priority].sla;
    return new Date(Date.now() + hours * 60 * 60 * 1000);
  }
  
  async updateTicket(ticketId, update) {
    const ticket = await this.getTicket(ticketId);
    
    // 添加消息
    if (update.message) {
      ticket.messages.push({
        from: update.from,
        content: update.message,
        time: new Date()
      });
    }
    
    // 更新状态
    if (update.status) {
      ticket.status = update.status;
      
      if (update.status === 'resolved') {
        ticket.resolvedAt = new Date();
        // 发送满意度调查
        await this.sendSatisfactionSurvey(ticket);
      }
    }
    
    await this.saveTicket(ticket);
    return ticket;
  }
}
```

```javascript
class ServiceQualityMonitor {
  async calculateAgentScore(agentId, period) {
    const metrics = await this.getAgentMetrics(agentId, period);
    
    const score = {
      // 响应效率 (30%)
      responseTime: this.scoreResponseTime(metrics.avgResponseTime),
      resolutionTime: this.scoreResolutionTime(metrics.avgResolutionTime),
      
      // 解决能力 (30%)
      firstContactResolution: metrics.fcr * 100,
      resolutionRate: (metrics.resolved / metrics.total) * 100,
      
      // 客户满意度 (25%)
      csat: metrics.csat * 20,
      nps: this.normalizeNPS(metrics.nps),
      
      // 工作量 (15%)
      ticketsHandled: this.scoreVolume(metrics.total),
      customerComplaints: Math.max(0, 100 - metrics.complaints * 10)
    };
    
    score.total = this.calculateWeightedAverage(score);
    
    return score;
  }
  
  async generateQualityReport(period) {
    const agents = await this.getAllAgents();
    const report = {
      period: period,
      summary: {
        totalTickets: 0,
        avgResponseTime: 0,
        avgCSAT: 0,
        avgFCR: 0
      },
      agents: [],
      topPerformers: [],
      improvementNeeded: []
    };
    
    for (const agent of agents) {
      const score = await this.calculateAgentScore(agent.id, period);
      report.agents.push({
        id: agent.id,
        name: agent.name,
        score: score
      });
      
      // 更新汇总
      report.summary.totalTickets += score.ticketsHandled;
    }
    
    // 排序
    report.agents.sort((a, b) => b.score.total - a.score.total);
    
    // 标记优秀和需改进
    report.topPerformers = report.agents.slice(0, 3);
    report.improvementNeeded = report.agents.slice(-3).reverse();
    
    return report;
  }
}
```
