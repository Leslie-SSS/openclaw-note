# 购物车优化策略

## 一、理论框架

### 1.1 购物车在转化漏斗中的关键位置

购物车是用户购买决策的最后一道关卡，也是弃单率最高的环节：

**购物车弃单数据洞察：**
- 全球平均购物车弃单率：69.82%
- 移动端弃单率更高：约85%
- 每年因弃单损失的销售额：约4.6万亿美元
- 优化购物车可挽回35-50%的弃单

**弃单原因分析：**
| 弃单原因 | 占比 | 优化方向 |
|---------|------|---------|
| 额外费用太高（运费、税费） | 48% | 运费透明、包邮门槛 |
| 需要创建账户 | 26% | 游客结账 |
| 结账流程太长 | 23% | 简化流程 |
| 无法计算总价 | 21% | 实时价格计算 |
| 不信任网站 | 18% | 信任元素展示 |
| 网站崩溃/错误 | 17% | 技术稳定性 |
| 支付方式不足 | 13% | 多样化支付 |

### 1.2 购物车心理学

### 1.4 购物车类型分析

**传统购物车：**
- 固定页面
- 完整商品列表
- 详细价格信息
- 适用：标准电商

**侧滑购物车（Slide-out Cart）：**
- 不离开当前页面
- 快速预览
- 一键结账
- 适用：快速购买场景

**迷你购物车（Mini Cart）：**
- 悬浮显示
- 简要信息
- 快速操作
- 适用：多页面浏览

**Ajax购物车：**
- 无刷新添加
- 实时更新
- 流畅体验
- 适用：现代电商

```javascript
// 数量调整逻辑
class CartQuantity {
  constructor(element) {
    this.element = element;
    this.input = element.querySelector('.qty-input');
    this.minusBtn = element.querySelector('.qty-minus');
    this.plusBtn = element.querySelector('.qty-plus');
    this.itemId = element.closest('.cart-item').dataset.itemId;
    
    this.bindEvents();
  }
  
  bindEvents() {
    this.minusBtn.addEventListener('click', () => this.updateQuantity(-1));
    this.plusBtn.addEventListener('click', () => this.updateQuantity(1));
    this.input.addEventListener('change', (e) => {
      this.setQuantity(parseInt(e.target.value));
    });
  }
  
  async updateQuantity(delta) {
    const newQty = parseInt(this.input.value) + delta;
    if (newQty >= 1 && newQty <= 10) {
      await this.setQuantity(newQty);
    }
  }
  
  async setQuantity(qty) {
    // 显示加载状态
    this.input.disabled = true;
    
    try {
      // 发送API请求
      const response = await fetch('/api/cart/update', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ itemId: this.itemId, quantity: qty })
      });
      
      if (response.ok) {
        this.input.value = qty;
        this.updateCartTotals(await response.json());
      }
    } catch (error) {
      // 显示错误提示
      this.showError('更新失败，请重试');
    } finally {
      this.input.disabled = false;
    }
  }
  
  updateCartTotals(data) {
    // 更新购物车总计
    document.querySelector('.total-amount').textContent = data.total;
    document.querySelector('.cart-count').textContent = data.itemCount;
    // 更新当前商品小计
    const itemTotal = this.element.closest('.cart-item')
      .querySelector('.item-total-price');
    itemTotal.textContent = data.itemTotal;
  }
}
```

```javascript
// 删除商品（带撤销功能）
class CartItemRemover {
  constructor() {
    this.removedItems = [];
    this.undoTimeout = 10000; // 10秒内可撤销
  }
  
  async removeItem(itemId) {
    const itemElement = document.querySelector(`[data-item-id="${itemId}"]`);
    const itemData = this.getItemData(itemElement);
    
    // 保存删除的项目
    this.removedItems.push({
      id: itemId,
      data: itemData,
      element: itemElement.cloneNode(true)
    });
    
    // 显示撤销提示
    this.showUndoToast(itemId);
    
    // 隐藏商品
    itemElement.style.opacity = '0';
    itemElement.style.height = '0';
    
    // 延迟后真正删除
    setTimeout(() => {
      if (!this.isItemRestored(itemId)) {
        this.permanentlyRemove(itemId);
      }
    }, this.undoTimeout);
  }
  
  showUndoToast(itemId) {
    const toast = document.createElement('div');
    toast.className = 'undo-toast';
    toast.innerHTML = `
      <span>已移除商品</span>
      <button class="undo-btn" data-item-id="${itemId}">撤销</button>
    `;
    document.body.appendChild(toast);
    
    toast.querySelector('.undo-btn').addEventListener('click', () => {
      this.restoreItem(itemId);
      toast.remove();
    });
    
    setTimeout(() => toast.remove(), this.undoTimeout);
  }
  
  async restoreItem(itemId) {
    const removedItem = this.removedItems.find(item => item.id === itemId);
    if (removedItem) {
      // 恢复UI
      const cartItems = document.querySelector('.cart-items');
      cartItems.appendChild(removedItem.element);
      
      // 恢复API状态
      await fetch('/api/cart/restore', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ itemId })
      });
      
      this.removedItems = this.removedItems.filter(item => item.id !== itemId);
    }
  }
}
```

```javascript
// 优惠码应用
class PromoCodeHandler {
  constructor() {
    this.input = document.getElementById('promoInput');
    this.applyBtn = document.getElementById('promoApply');
    this.appliedPromos = [];
    
    this.bindEvents();
  }
  
  bindEvents() {
    this.applyBtn.addEventListener('click', () => this.applyPromo());
    this.input.addEventListener('keypress', (e) => {
      if (e.key === 'Enter') this.applyPromo();
    });
  }
  
  async applyPromo() {
    const code = this.input.value.trim().toUpperCase();
    
    if (!code) {
      this.showError('请输入优惠码');
      return;
    }
    
    if (this.appliedPromos.includes(code)) {
      this.showError('此优惠码已应用');
      return;
    }
    
    this.applyBtn.disabled = true;
    this.applyBtn.textContent = '验证中...';
    
    try {
      const response = await fetch('/api/promo/validate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ code })
      });
      
      const result = await response.json();
      
      if (result.valid) {
        this.applyPromoSuccess(code, result);
      } else {
        this.showError(result.message || '无效的优惠码');
      }
    } catch (error) {
      this.showError('验证失败，请重试');
    } finally {
      this.applyBtn.disabled = false;
      this.applyBtn.textContent = '应用';
    }
  }
  
  applyPromoSuccess(code, result) {
    this.appliedPromos.push(code);
    this.input.value = '';
    
    // 显示应用成功的优惠码
    const promoList = document.querySelector('.applied-promos');
    promoList.innerHTML += `
      <div class="promo-item">
        <span class="promo-code">${code}</span>
        <span class="promo-discount">-¥${result.discount}</span>
        <button class="promo-remove" data-code="${code}">×</button>
      </div>
    `;
    
    // 更新总价
    this.updateTotal(result);
  }
}
```

```javascript
// 自动保存购物车
class CartPersistence {
  constructor() {
    this.storageKey = 'saved_cart';
    this.autoSaveInterval = 30000; // 30秒自动保存
    this.init();
  }
  
  init() {
    // 页面加载时恢复购物车
    this.restoreCart();
    
    // 定期保存
    setInterval(() => this.saveCart(), this.autoSaveInterval);
    
    // 离开页面前保存
    window.addEventListener('beforeunload', () => this.saveCart());
  }
  
  saveCart() {
    const cartData = this.getCartData();
    localStorage.setItem(this.storageKey, JSON.stringify({
      items: cartData,
      savedAt: Date.now()
    }));
  }
  
  restoreCart() {
    const saved = localStorage.getItem(this.storageKey);
    if (saved) {
      const { items, savedAt } = JSON.parse(saved);
      // 如果保存时间超过7天，清除
      if (Date.now() - savedAt > 7 * 24 * 60 * 60 * 1000) {
        localStorage.removeItem(this.storageKey);
      } else {
        this.showRestoredCart(items);
      }
    }
  }
  
  showRestoredCart(items) {
    // 显示恢复购物车的提示
    const toast = document.createElement('div');
    toast.className = 'cart-restore-toast';
    toast.innerHTML = `
      <span>你有未完成的购物车 (${items.length}件商品)</span>
      <button class="restore-btn">恢复购物车</button>
      <button class="dismiss-btn">忽略</button>
    `;
    document.body.appendChild(toast);
  }
}
```

```javascript
      <div class="bundle-items">
        ${products.map(p => `
          <div class="bundle-item">
            <img src="${p.image}" alt="${p.name}">
            <span class="bundle-item-name">${p.name}</span>
            <span class="bundle-item-price">¥${p.price}</span>
          </div>
        `).join(' + ')}
      </div>
      <div class="bundle-savings">
        套装优惠价: ¥${bundlePrice} (省¥${savings})
      </div>
    `;
  }
}
```
