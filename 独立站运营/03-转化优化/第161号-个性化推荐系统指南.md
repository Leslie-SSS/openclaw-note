## 二、推荐算法详解

### 2.1 协同过滤算法

```
协同过滤 (Collaborative Filtering):

1. 基于用户的协同过滤 (User-based CF)
   
   原理: 找到相似用户，推荐他们喜欢的商品
   
   相似度计算:
   - 余弦相似度
   - 皮尔逊相关系数
   - Jaccard相似度
   
   公式:
   sim(u,v) = Σ(r_ui - r̄_u)(r_vi - r̄_v) / √[Σ(r_ui - r̄_u)² × Σ(r_vi - r̄_v)²]
   
   推荐:
   r_ui = r̄_u + Σ sim(u,v) × (r_vi - r̄_v) / Σ|sim(u,v)|

2. 基于物品的协同过滤 (Item-based CF)
   
   原理: 推荐与用户历史喜好商品相似的商品
   
   相似度计算:
   sim(i,j) = |U_i ∩ U_j| / √(|U_i| × |U_j|)
   
   优势:
   - 物品相似度相对稳定，可预计算
   - 可解释性强
   - 适合商品数量稳定的场景

3. 矩阵分解 (Matrix Factorization)
   
   R ≈ U × V^T
   
   其中:
   - R: 用户-物品评分矩阵
   - U: 用户隐因子矩阵
   - V: 物品隐因子矩阵
   
   损失函数:
   L = Σ(r_ui - u_i · v_j)² + λ(||u_i||² + ||v_j||²)
   
   求解: SVD, ALS, SGD
```

### 2.2 内容推荐算法

```python
# 内容推荐算法示例

import numpy as np
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

class ContentBasedRecommender:
    """基于内容的推荐"""
    
    def __init__(self):
        self.tfidf = TfidfVectorizer()
        self.item_features = None
        self.item_ids = None
    
    def fit(self, items):
        """
        items: list of {id, title, description, category, tags}
        """
        self.item_ids = [item['id'] for item in items]
        
        # 构建文本特征
        texts = [
            f"{item['title']} {item['description']} {' '.join(item['tags'])}"
            for item in items
        ]
        
        # TF-IDF向量化
        self.item_features = self.tfidf.fit_transform(texts)
        
        return self
    
    def get_similar_items(self, item_id, top_k=10):
        """获取相似商品"""
        idx = self.item_ids.index(item_id)
        item_vector = self.item_features[idx]
        
        # 计算余弦相似度
        similarities = cosine_similarity(item_vector, self.item_features)[0]
        
        # 获取top_k相似商品（排除自己）
        similar_indices = np.argsort(similarities)[::-1][1:top_k+1]
        
        return [
            (self.item_ids[i], similarities[i])
            for i in similar_indices
        ]
    
    def recommend_for_user(self, user_history, top_k=10):
        """为用户推荐"""
        # 获取用户历史商品的索引
        history_indices = [
            self.item_ids.index(item_id)
            for item_id in user_history
            if item_id in self.item_ids
        ]
        
        if not history_indices:
            return []
        
        # 用户画像 = 历史商品特征的平均
        user_profile = np.mean(
            self.item_features[history_indices].toarray(),
            axis=0
        )
        
        # 计算与所有商品的相似度
        similarities = cosine_similarity(
            user_profile.reshape(1, -1),
            self.item_features
        )[0]
        
        # 排除已购买商品
        for idx in history_indices:
            similarities[idx] = -1
        
        # 获取推荐
        recommend_indices = np.argsort(similarities)[::-1][:top_k]
        
        return [
            (self.item_ids[i], similarities[i])
            for i in recommend_indices
        ]
```

```python
# 深度学习推荐模型示例

import torch
import torch.nn as nn

class NCF(nn.Module):
    """Neural Collaborative Filtering"""
    
    def __init__(self, num_users, num_items, embedding_dim=64, layers=[128, 64, 32]):
        super(NCF, self).__init__()
        
        # GMF部分
        self.user_embedding_gmf = nn.Embedding(num_users, embedding_dim)
        self.item_embedding_gmf = nn.Embedding(num_items, embedding_dim)
        
        # MLP部分
        self.user_embedding_mlp = nn.Embedding(num_users, embedding_dim)
        self.item_embedding_mlp = nn.Embedding(num_items, embedding_dim)
        
        # MLP层
        mlp_layers = []
        input_size = embedding_dim * 2
        for layer_size in layers:
            mlp_layers.append(nn.Linear(input_size, layer_size))
            mlp_layers.append(nn.ReLU())
            mlp_layers.append(nn.Dropout(0.2))
            input_size = layer_size
        self.mlp = nn.Sequential(*mlp_layers)
        
        # 输出层
        self.output = nn.Linear(embedding_dim + layers[-1], 1)
        self.sigmoid = nn.Sigmoid()
    
    def forward(self, user_ids, item_ids):
        # GMF
        user_gmf = self.user_embedding_gmf(user_ids)
        item_gmf = self.item_embedding_gmf(item_ids)
        gmf_output = user_gmf * item_gmf
        
        # MLP
        user_mlp = self.user_embedding_mlp(user_ids)
        item_mlp = self.item_embedding_mlp(item_ids)
        mlp_input = torch.cat([user_mlp, item_mlp], dim=-1)
        mlp_output = self.mlp(mlp_input)
        
        # 合并
        combined = torch.cat([gmf_output, mlp_output], dim=-1)
        output = self.sigmoid(self.output(combined))
        
        return output.squeeze()


class DIN(nn.Module):
    """Deep Interest Network for Click-Through Rate Prediction"""
    
    def __init__(self, num_features, embedding_dim=16):
        super(DIN, self).__init__()
        
        self.embeddings = nn.ModuleDict({
            name: nn.Embedding(num, embedding_dim)
            for name, num in num_features.items()
        })
        
        # 注意力机制
        self.attention = nn.Sequential(
            nn.Linear(embedding_dim * 4, 64),
            nn.ReLU(),
            nn.Linear(64, 1),
            nn.Softmax(dim=1)
        )
        
        # MLP
        self.mlp = nn.Sequential(
            nn.Linear(embedding_dim * len(num_features), 128),
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, 1),
            nn.Sigmoid()
        )
    
    def attention_pooling(self, candidate, history):
        """
        candidate: [batch, dim]
        history: [batch, seq_len, dim]
        """
        batch_size, seq_len, dim = history.shape
        
        # 扩展候选向量
        candidate = candidate.unsqueeze(1).expand(-1, seq_len, -1)
        
        # 注意力输入
        att_input = torch.cat([
            history,
            candidate,
            history * candidate,
            history - candidate
        ], dim=-1)
        
        # 计算注意力权重
        att_weights = self.attention(att_input)
        
        # 加权求和
        pooled = torch.sum(att_weights * history, dim=1)
        
        return pooled
```

```python
class RecallLayer:
    """多路召回"""
    
    def __init__(self):
        self.cf_recaller = CFRecaller()
        self.content_recaller = ContentRecaller()
        self.hot_recaller = HotRecaller()
        self.recent_recaller = RecentRecaller()
    
    def recall(self, user_id, context, top_k=1000):
        """
        多路召回
        返回: [(item_id, score, source)]
        """
        candidates = []
        
        # 1. 协同过滤召回
        cf_items = self.cf_recaller.recall(user_id, top_k=300)
        candidates.extend([(i, s, 'cf') for i, s in cf_items])
        
        # 2. 内容召回
        content_items = self.content_recaller.recall(user_id, top_k=200)
        candidates.extend([(i, s, 'content') for i, s in content_items])
        
        # 3. 热门召回
        hot_items = self.hot_recaller.recall(context['category'], top_k=200)
        candidates.extend([(i, s, 'hot') for i, s in hot_items])
        
        # 4. 最近行为召回
        recent_items = self.recent_recaller.recall(user_id, top_k=100)
        candidates.extend([(i, s, 'recent') for i, s in recent_items])
        
        # 去重合并
        item_scores = {}
        for item_id, score, source in candidates:
            if item_id not in item_scores:
                item_scores[item_id] = {'score': 0, 'sources': []}
            item_scores[item_id]['score'] += score
            item_scores[item_id]['sources'].append(source)
        
        # 排序返回
        sorted_items = sorted(
            item_scores.items(),
            key=lambda x: x[1]['score'],
            reverse=True
        )[:top_k]
        
        return [(i, s['score'], s['sources']) for i, s in sorted_items]
```

```python
# 排序层实现

class RankingLayer:
    """精排层"""
    
    def __init__(self, model_path):
        self.model = self.load_model(model_path)
        self.feature_extractor = FeatureExtractor()
    
    def rank(self, user_id, candidates, context, top_k=100):
        """
        对候选商品进行精细排序
        """
        # 1. 特征提取
        features = []
        for item_id, recall_score, sources in candidates:
            feature = self.feature_extractor.extract(user_id, item_id, context)
            feature['recall_score'] = recall_score
            feature['recall_sources'] = len(sources)
            features.append((item_id, feature))
        
        # 2. 批量预测
        item_ids = [f[0] for f in features]
        feature_matrix = self.prepare_features([f[1] for f in features])
        scores = self.model.predict(feature_matrix)
        
        # 3. 排序
        ranked = sorted(
            zip(item_ids, scores),
            key=lambda x: x[1],
            reverse=True
        )[:top_k]
        
        return ranked
```

```python
        """
        重排: 多样性 + 新鲜度 + 业务规则
        """
        items = list(ranked_items)
        
        # 1. 应用业务规则
        items = self.apply_business_rules(items, user_id, context)
        
        # 2. 多样性重排
        items = self.diversity_rerank(items)
        
        # 3. 新鲜度调整
        items = self.freshness_adjust(items)
        
        return items
    
    def diversity_rerank(self, items, k=5):
        """多样性重排 (MMR算法)"""
        selected = []
        remaining = list(items)
        
        while len(selected) < min(k, len(items)):
            if not selected:
                # 第一个选相关性最高的
                selected.append(remaining.pop(0))
            else:
                # MMR选择
                best_score = -float('inf')
                best_idx = 0
                
                for i, (item_id, score) in enumerate(remaining):
                    # 计算与已选商品的相似度
                    max_sim = max(
                        self.similarity(item_id, s[0])
                        for s in selected

                    # MMR分数

                        (1 - self.diversity_weight) * score -
                        self.diversity_weight * max_sim
                    )
                    
                    if mmr_score > best_score:
                        best_score = mmr_score

        selected.extend(remaining)
        
        return selected
```

## 五、推荐效果评估

### 5.1 离线评估指标

```yaml
离线评估指标:

准确率指标:
  Precision@K:
    定义: Top-K推荐中用户喜欢的比例
    公式: |推荐命中 ∩ 用户喜欢| / K
  
  Recall@K:
    定义: 用户喜欢商品被推荐的比例
    公式: |推荐命中 ∩ 用户喜欢| / |用户喜欢|
  
  F1@K:
    定义: Precision和Recall的调和平均
    公式: 2 × P × R / (P + R)

排序质量指标:
  NDCG@K (Normalized Discounted Cumulative Gain):
    定义: 考虑位置的排序质量
    公式: DCG@K / IDCG@K
    
    DCG@K = Σ (2^rel_i - 1) / log2(i + 1)
    
  MAP (Mean Average Precision):
    定义: 平均精度均值
    公式: Σ P@k × rel(k) / |相关文档|
  
  MRR (Mean Reciprocal Rank):
    定义: 第一个正确答案的位置倒数
    公式: 1 / rank(first_correct)

覆盖率指标:
  Coverage:
    定义: 被推荐商品占总商品比例
    
  Diversity:
    定义: 推荐列表的多样性
    
  Novelty:
    定义: 推荐商品的新颖度
```

### 5.2 在线评估指标


```python
# 推荐系统A/B测试

class RecommenderABTest:
    """推荐系统A/B测试"""
    
    def __init__(self):
        self.experiments = {}
    
    def get_model(self, user_id, experiment_id):
        """获取用户应使用的模型"""
        # 一致性分流
        bucket = hash(f"{user_id}_{experiment_id}") % 100
        
        if bucket < 50:
            return 'control'  # 对照组
        else:
            return 'treatment'  # 实验组
    
    def recommend(self, user_id, context):
        """根据分组返回推荐结果"""
        experiment_id = 'rec_model_v2'
        model_name = self.get_model(user_id, experiment_id)
        
        if model_name == 'control':
            items = self.control_model.recommend(user_id, context)
        else:
            items = self.treatment_model.recommend(user_id, context)
        
        # 记录日志
        self.log_recommend(user_id, model_name, items, experiment_id)
        
        return items
    
    def analyze_results(self, experiment_id, start_date, end_date):
        """分析实验结果"""
        # 获取数据
        control_data = self.get_metrics(experiment_id, 'control', start_date, end_date)
        treatment_data = self.get_metrics(experiment_id, 'treatment', start_date, end_date)
        
        # 计算指标
        metrics = {
            'ctr': {
                'control': control_data['clicks'] / control_data['impressions'],
                'treatment': treatment_data['clicks'] / treatment_data['impressions'],
            },
            'cvr': {
                'control': control_data['conversions'] / control_data['clicks'],
                'treatment': treatment_data['conversions'] / treatment_data['clicks'],
            },
            'gmv_per_user': {
                'control': control_data['gmv'] / control_data['users'],
                'treatment': treatment_data['gmv'] / treatment_data['users'],
            }
        }
        
        # 显著性检验
        for metric_name, values in metrics.items():
            p_value = self.significance_test(
                values['control'],
                values['treatment'],
                control_data['sample_size'],
                treatment_data['sample_size']
            )
            metrics[metric_name]['p_value'] = p_value
            metrics[metric_name]['significant'] = p_value < 0.05
        
        return metrics
```

```yaml
      need_amount = threshold - sum(item.price for item in cart_items)
      
      if need_amount > 0:
```
